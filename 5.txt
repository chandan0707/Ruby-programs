class Palindrome
def cal
rev=0
puts "enter a Number"
a=gets.to_i
temp=a
while a>0
d=a%10
rev=(rev*10)+d
a=a/10
end
puts"The number you entered is #{ temp}"
puts "The reverse of the number is #{ rev}"

if temp== rev
puts"hey!!Its a palindrome"

else
puts"The string of number entered by you is Not a palindrome"
end
end
end
Palindrome.new.cal
